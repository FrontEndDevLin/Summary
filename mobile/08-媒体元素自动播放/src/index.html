08-媒体元素自动播放


  触屏即播
  
  let videoObject = {
    el: document.getElementById("video"),
    onload: false,
    play: false
  }
  
  videoObject.el.oncanplay = () => {
    videoObject.onload = true;
  }
  
  document.addEventListener("touchend", () => {
    if (!videoObject.play) {
      videoObject.el.play().then(() => {
        alert("自动播放")
      }).catch((err) => {
        alert(err);
      });
      videoObject.play = true;
    }
  })
